Adding events

Adding event listeners

To make sure that the browser calls a particualar function when an event happens on an element, you must use document.addEventListener: 

var buttonEl = document.getElementById("clicker");
var onButtonClick = function() {
    console.log("Oh nice, you clicked me!");
};
buttonEl.addEventListener("click", onButtonClick);

Examples

Changing content of a button after a click.

<button id="clicker">Click and see the difference</button>
<script>
var clickerButton = document.getElementById("clicker");
var onButtonClick = function () { 
    clickerButton.textContent =     "You clicked me!";
}
clickerButton.addEventListener("click", onButtonClick);
</script>

Processing forms with events

</label><br>
<label>What language do you speak?
  <select id="lang>
    <option value="en">English</option>
    <option value="pl">Polish</option>
    <option value="es">Spanish</option>
  </select>
</label><br>
<button id="button" type="button">Click me!</button>
<div id="message"></div>
<script>
var button = document.getElementById("button");
var formButtonClick = function() {
        document.getElementById("message").textContent = "";
        var name = document.getElementById("name").value;
        var lang = document.getElementById("lang").value;
        var greeting;
        if (lang === "en") {
                greeting = "Hey " + name;
        } else if (lang === "pl") {
                greeting = "Cześć " + name;
        } else {
                greeting = "Hola " + name;
        }
        document.getElementById("message").textContent += greeting;
};
button.addEventListener("click", formButtonClick);
<script>
